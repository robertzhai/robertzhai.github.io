# sed 替换数字为空
	sed -r 's/[0-9]+//g' data.txt
	
# sed 不考虑特殊字符进行替换
	sed -e 's#user_id`#user_id#g' data.txt
	
# sed 正则替换
	sed -r "s/user_id='(.*?)'/user_id=''/g" > data.txt
	
# awk 指定多个分割字符
	cat  update.txt  | awk -F 'QUERY|SET|WHERE' '{print $1,"where" $3;}' >update_filter.txt
